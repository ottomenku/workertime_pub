Facade:--------------------------------------------

osztály elkészítése : App\Handlers\CalendarHandler;

service provider elkészítése :App\Providers

namespace App\Providers;
use Illuminate\Support\ServiceProvider;
use Illuminate\Support\Facades\App;

class CalendarHandlerServiceProvider extends ServiceProvider
{
    public function boot(){}
    public function register()
    {
        App::bind('CalendarHandler', function()
        {
            return new \App\Handlers\CalendarHandler;
        });
    }

}
facade elkészítése

 namespace App\Facades;
class CalendarHandler extends \Illuminate\Support\Facades\Facade
{
    public static function getFacadeAccessor()
    {
        return 'CalendarHandler';
    }
}

config\app:

    'providers' => [  App\Providers\CalendarHandlerServiceProvider::class ],

    'aliases' => [ 'CalendarHandler'=> App\Facades\CalendarHandler::class, ]
